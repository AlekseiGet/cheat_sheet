вычисления на основе массива какого-либо единого значения (с числами)

let numValue = num.reduce(function(previosValue, item, index, array){
return item + previosValue
},0);
console.log(numValue);
previosValue - это результата дийствия прошлой итерации цикла
0 - это чему равен previosValue в первый раз, если не подставить "0", то он будер равен первому значению

вычисление единого значения на основе всего массива. Выведет весь массив записаный в строку

let arrValue = arr.reduce(function(previosValue, item, index, array) {
return `${previosValue}, ${item}`;
});
console.log(arrValue);
