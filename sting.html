соединение строк

`${text}, ${name}!`

|| Или - true


console.log(1 || 0); //1
console.log(true || "Alexsei"); //true
console.log(null || 56); // 56
console.log(null || "Alexsei"); //Alexsei
console.log(undefined || '' || null || 0); //0

&& И - false


console.log('aleksei' && 0 && 2);//0
console.log(1 && 2 && null && 3);//null

! Не (отрицание)

console.log(!true && 65 || 33 && !1);//false

?? сравнивает с null и возвращает оперантд если он не null или undefined


console.log(name ?? 'нет имени' ?? '');//нет имени
